### 1. 1009

```js
const readline = require("readline");

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
});

let result = 0;
let count = 0;
let numberList = [];

rl.on("line", function (line) {
  if (!count) {
    count = Number(line);
  } else {
    numberList.push(line.split(" "));
  }

  if (numberList.length === count) {
    for (let i = 0; i < numberList.length; i++) {
      let a = numberList[i][0];
      let b = numberList[i][1];
      result = 1;

      for (let i = 0; i < b; i++) {
        result = (result * a) % 10;
        if (!result) {
          result = 10;
        }
      }
      console.log(result);
    }
    rl.close();
  }
}).on("close", function () {
  process.exit();
});
```

### 2. 1076

```js
const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
});

let inputList = [];
let result;
const colors = [
  "black",
  "brown",
  "red",
  "orange",
  "yellow",
  "green",
  "blue",
  "violet",
  "grey",
  "white",
];

rl.on("line", function (line) {
  inputList.push(line);
  const index = inputList.map((item) => colors.indexOf(item));
  const numString = index[0] + "" + index[1];
  result = Number(numString) * Math.pow(10, index[2]);
}).on("close", function () {
  console.log(result);
  process.exit();
});
```
